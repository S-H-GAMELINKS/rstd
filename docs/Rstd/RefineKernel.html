refine Kernel do
  def import(path, *import_classes)
    raise "Kernel#import is not supported" if RUBY_VERSION > "2.4" && "2.7" > RUBY_VERSION 
    require path

    klasses = []

    import_classes.each do |klass|
      klasses.unshift Object.const_get(klass)
    end

    return klasses.reverse
  end
end
